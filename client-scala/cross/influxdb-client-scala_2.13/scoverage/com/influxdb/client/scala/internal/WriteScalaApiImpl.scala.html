46    * Write Line Protocol record into specified bucket.
47    *
48    * @param precision Precision for the unix timestamps within the body line-protocol.
49    *                  The [[com.influxdb.client.domain.WritePrecision.NS]] will be used as the precision if not specified.
50    * @param bucket    Specifies the destination bucket for writes.
51    *                  The [[com.influxdb.client.InfluxDBClientOptions#getBucket]] will be used as the destination
52    *                  `bucket` if the `bucket` is not specified.
53    * @param org       Specifies the destination organization for writes.
54    *                  The [[com.influxdb.client.InfluxDBClientOptions#getOrg]] will be used as the destination `organization`
55    *                  if the `org` is not specified.
56    * @return the sink that accept the record specified in InfluxDB Line Protocol. The `record` is considered as one batch unit.
57    */
58   override def writeRecord(precision: Option[WritePrecision], bucket: Option[String], org: Option[String]): Sink[String, Future[Done]] = {
59     Flow[String]
60       .map(record => Seq(new AbstractWriteClient.BatchWriteDataRecord(record)))
61       .map(batch => writeHttp(precision, bucket, org, batch))
62       .toMat(Sink.head)(Keep.right)
63   }

128   /**
129    * Write Data points into specified bucket.
130    *
131    * @param parameters specify InfluxDB Write endpoint parameters
132    * @return the sink that accept the Data points. The `points` are considered as one batch unit.
133    */
134   override def writePoints(parameters: WriteParameters): Sink[Seq[Point], Future[Done]] = {
135     Flow[Seq[Point]]
136       // create ordered Map<Precision, Seq[Point]>
137       .map(records => records.foldRight(ListMap.empty[WritePrecision, Seq[Point]]) {
138         case (point, map) => map.updated(point.getPrecision, point +: map.getOrElse(point.getPrecision, Seq()))
139       }.toList.reverse)
140       .map(grouped => grouped.map(group => (group._1, group._2.map(point => new AbstractWriteClient.BatchWriteDataPoint(point, options)))))
141       .map(batches => batches.foreach(batch => writeHttp(parameters.copy(batch._1, options), batch._2)))
142       .map(_ => Done.done())
143       .toMat(Sink.head)(Keep.right)
144   }

160   override def writeMeasurement[M](precision: Option[WritePrecision], bucket: Option[String], org: Option[String]): Sink[M, Future[Done]] = {
161     Flow[M]
162       .map(measurement => {
163         val parameters = toWriteParameters(precision, bucket, org)
164         Seq(toMeasurementBatch(measurement, parameters.precisionSafe(options)))
165       })
166       .map(batch => writeHttp(precision, bucket, org, batch))
167       .toMat(Sink.head)(Keep.right)
168   }

193    * @return the sink that accept the measurements. The `measurements` are considered as one batch unit.

195   override def writeMeasurements[M](parameters: WriteParameters): Sink[Seq[M], Future[Done]] = {
196     Flow[Seq[M]]

198       .map(batch => writeHttp(parameters, batch))
199       .toMat(Sink.head)(Keep.right)
200   }

202   private def writeHttp(precision: Option[WritePrecision], bucket: Option[String], org: Option[String], batch: Seq[AbstractWriteClient.BatchWriteData]): Done = {
203     writeHttp(toWriteParameters(precision, bucket, org), batch)
204   }
205 
206   private def writeHttp(parameters: WriteParameters, batch: Seq[AbstractWriteClient.BatchWriteData]): Done = {
207     write(parameters, batch.toList.asJava.stream())
208     Done.done()
209   }
210 
211   private def toWriteParameters(precision: Option[WritePrecision], bucket: Option[String], org: Option[String]): WriteParameters = {
212     val parameters = new WriteParameters(bucket.orNull, org.orNull, precision.orNull, null)
213     parameters.check(options)
214     parameters
215   }
