37   /**
38    * Get the Query client.
39    *
40    * @return the new client instance for the Query API
41    */
42   override def getQueryScalaApi(): QueryScalaApi = new QueryScalaApiImpl(retrofit.create(classOf[QueryService]), options)
43 
44 
45   /**
46    * Create a new WriteApi client.
47    *
48    * @return the new client instance for the Write API
49    */
50   override def getWriteScalaApi: WriteScalaApi = new WriteScalaApiImpl(retrofit.create(classOf[WriteService]), options)
51 
52   /**
53    * Get the health of an instance.
54    *
55    * @return health of an instance
56    */

58 
59   /**
60    * Check the status of InfluxDB Server.
61    *
62    * @return true if server is healthy otherwise return false
63    */
64   override def ping: Boolean = ping(pingService.getPing)
65 
66   /**
67    * Returns the version of the connected InfluxDB Server.
68    *
69    * @return the version String, otherwise unknown.
70    */
71   override def version: String = version(pingService.getPing)
72 
73   /**
74    * Gets the [[LogLevel]] that is used for logging requests and responses.
75    *
76    * @return the [[LogLevel]] that is used for logging requests and responses

78   override def getLogLevel: LogLevel = getLogLevel(loggingInterceptor)
79 
80   /**
81    * Sets the log level for the request and response information.
82    *
83    * @param logLevel the log level to set.
84    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
85    */
86   override def setLogLevel(logLevel: LogLevel): InfluxDBClientScala = {
87     setLogLevel(loggingInterceptor, logLevel)
88     this
89   }
90 
91   /**
92    * Enable Gzip compress for http requests.
93    *
94    * Currently only the "Write" and "Query" endpoints supports the Gzip compression.
95    *
96    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
97    */
98   override def enableGzip(): InfluxDBClientScala = {
99     gzipInterceptor.enableGzip()
100     this
101   }
102 
103   /**
104    * Disable Gzip compress for http request body.
105    *
106    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
107    */
108   override def disableGzip(): InfluxDBClientScala = {
109     gzipInterceptor.disableGzip()
110     this
111   }
112 
113   /**
114    * Returns whether Gzip compress for http request body is enabled.
115    *
116    * @return true if gzip is enabled.
117    */
118   override def isGzipEnabled: Boolean = gzipInterceptor.isEnabledGzip
