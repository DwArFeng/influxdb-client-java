30 /**
31  * The Factory that creates an instance of a Flux client.
32  * 
33  * @author Jakub Bednar (bednar@github) (05/11/2018 10:10)
34  */
35 object InfluxDBClientScalaFactory {
36 
37 
38   /**
39    * Create an instance of the InfluxDB 2.x client that is configured via `influx2.properties`.
40    * The `influx2.properties` has to be located on classpath.
41    *
42    * @return client
43    */
44   @Nonnull def create(): InfluxDBClientScala = {
45 
46     val options = InfluxDBClientOptions.builder()
47       .loadProperties()
48       .build()
49 
50     create(options)
51   }
52 
53   /**
54    * Create an instance of the InfluxDB 2.x client.
55    *
56    * The url could be a connection string with various configurations. For more info
57    * see: [[InfluxDBClientOptions.Builder#connectionString(connectionString:java.lang.String)]].
58    *
59    * @param connectionString connection string with various configurations.
60    * @return client
61    */
62   @Nonnull def create(connectionString: String): InfluxDBClientScala = {
63 
64     val options = InfluxDBClientOptions.builder()
65       .url(connectionString)
66       .build()
67 
68     create(options)
69   }
70 
71   /**
72    * Create an instance of the InfluxDB 2.x reactive client.
73    *
74    * The ''username/password'' auth is based on
75    * [[http://bit.ly/http-basic-auth HTTP "Basic" authentication]]. The authorization expires when the
76    * [[http://bit.ly/session-length time-to-live (TTL)]] (default 60 minutes) is reached
77    * and client produces [[com.influxdb.exceptions.UnauthorizedException]].
78    *
79    * The url could be a connection string with various configurations. For more info
80    * see: [[InfluxDBClientOptions.Builder#connectionString(connectionString:java.lang.String)]].
81    *
82    * @param url      the url the url to connect to InfluxDB (required). Example: http://localhost:8086?readTimeout=5000
83    * @param username the username to use in the basic auth
84    * @param password the password to use in the basic auth
85    * @return client
86    * @see [[InfluxDBClientOptions.Builder]]
87    */
88   @Nonnull def create(url: String, username: String, password: Array[Char]): InfluxDBClientScala = {
89 
90     val options = InfluxDBClientOptions.builder()
91       .url(url)
92       .authenticate(username, password)
93       .build()
94 
95     create(options)
96   }
97 
98   /**
99    * Create an instance of the InfluxDB 2.x reactive client.
100    *
101    * The url could be a connection string with various configurations. For more info
102    * see: [[InfluxDBClientOptions.Builder#connectionString(connectionString:java.lang.String)]].
103    *
104    * @param url   the url the url to connect to InfluxDB (required). Example: http://localhost:8086?readTimeout=5000
105    * @param token the token to use for the authorization
106    * @return client
107    * @see [[InfluxDBClientOptions.Builder]]
108    */
109   @Nonnull def create(url: String, token: Array[Char]): InfluxDBClientScala = {
110 
111     create(url, token, null)
112   }
113 
114   /**
115    * Create an instance of the InfluxDB 2.x reactive client.
116    *
117    * The url could be a connection string with various configurations. For more info
118    * see: [[InfluxDBClientOptions.Builder#connectionString(connectionString:java.lang.String)]].
119    *
120    * @param url   the url the url to connect to InfluxDB (required). Example: http://localhost:8086?readTimeout=5000
121    * @param token the token to use for the authorization
122    * @param org      the name of an organization
123    * @return client
124    * @see [[InfluxDBClientOptions.Builder]]
125    */
126   @Nonnull def create(url: String, token: Array[Char], org: String): InfluxDBClientScala = {
127 
128     create(url, token, org, null)
129   }
130 
131   /**
132    * Create an instance of the InfluxDB 2.x reactive client.
133    *
134    * The url could be a connection string with various configurations. For more info
135    * see: [[InfluxDBClientOptions.Builder#connectionString(connectionString:java.lang.String)]].
136    *
137    * @param url   the url the url to connect to InfluxDB (required). Example: http://localhost:8086?readTimeout=5000
138    * @param token the token to use for the authorization
139    * @param org      the name of an organization
140    * @param bucket   the name of a bucket
141    * @return client
142    * @see [[InfluxDBClientOptions.Builder]]
143    */
144   @Nonnull def create(url: String, token: Array[Char], org: String, bucket: String): InfluxDBClientScala = {
145 
146     val options = InfluxDBClientOptions.builder()
147       .url(url)
148       .authenticateToken(token)
149       .org(org)
150       .bucket(bucket)
151       .build()
152 
153     create(options)
154   }
155 
156   /**
157    * Create an instance of the InfluxDB 2.x reactive client.
158    *
159    * @param options the connection configuration
160    * @return client
161    */
162   @Nonnull def create(@Nonnull options: InfluxDBClientOptions): InfluxDBClientScala = {
163 
164     Arguments.checkNotNull(options, "InfluxDBClientOptions")
165 
166     new InfluxDBClientScalaImpl(options)
167   }
